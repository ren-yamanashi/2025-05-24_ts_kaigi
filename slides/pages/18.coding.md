```ts
import { Rule } from "eslint";

export const requireIfBlock: Rule.RuleModule = {
  meta: { /** ... */ },
  create(context) {
    return {
      IfStatement(node) {
        if (node.consequent.type !== "BlockStatement") {
          context.report({
            node,
            messageId: "requireIfBlock",
          });
        }
      },
    };
  },
};
```
