* ESLint から提供されるインターフェースに沿ってルールを書く

</div>

````md magic-move

```ts{*}
import { Rule } from "eslint";

export const sampleRule: Rule.RuleModule = {
  meta: {
    // ルールのメタデータを記述
  },
  create(context) {
    // ルールを記述
  },
};
```

```ts{8-10}
import { Rule } from "eslint";

export const sampleRule: Rule.RuleModule = {
  meta: {
    // ルールのメタデータを記述
  },
  create(context) {
    VariableDeclarator(node) {
      console.log(node) // AST に対する操作      
    }
  },
};
```
````
